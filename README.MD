```python
def mnmx(nums: list[float | int]):
    if not nums:  
        print('Value Error')
        return None  
    mn = min(nums)
    mx = max(nums)
    return (mn, mx)
    
def unique_sorted (nums: list[float | int]):
    return list( sorted( set(nums)))

def flatten(mat: list[list | tuple]):
    flattened_list = []
    for row in mat:
        if isinstance(row, (list, tuple)):
            flattened_list.extend(row)
        else:
            raise TypeError
    return flattened_list
```
